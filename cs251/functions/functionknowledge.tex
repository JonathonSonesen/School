\section*{1.)Proof:\\\mid f:~\mathbb{R}  \rightarrow \mathbb{R} \mid f(x) = \frac{(x^2)!}{x^2 + 10}~is~not~an~injective~function}
\begin{array}{l l l}
Row & Statement & Comment \\
1 &\exists x_1, x_2 \in \mathbb{R} \mid f(x_1) = f(x_2) \land x_1 \neq x_2 & \text{}\\
2 &  x_1 = 1 & \text{counter-example}&\\
 3& f(1) = \frac{(1^2)!}{(1)^2 + 10} = \frac{1}{11} & \text{subs.~to~premise}\\
4 & x_2 = -1 &\\
5 & f(-1) = \frac{(-1^2)!}{(-1)^2 + 10} = \frac{1}{11} & \text{subs.~to~premise}\\
7 & f(1) = f(-1) & \\
8 & x_1 \neq x_2 &\\
9 & \therefore \exists x_1, x_2 \in \mathbb{R} \mid f(x_1) = f(x_2) \land x_1 \neq x_2  &\\
\end{array}
