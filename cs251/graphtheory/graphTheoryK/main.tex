\begin{problem}[2]
  Let $G$ be the graph:
  \begin{center}  
  \begin{tikzpicture}[scale=1,transform shape]

      \GraphInit[vstyle=Dijkstra]
      \SetVertexMath
      \Vertex[x=0,y=0]{n_1}
      \Vertex[x=2,y=0]{n_2}
      \Vertex[x=0,y=-1]{n_3}
      \Vertex[x=2,y=-1]{n_4}
      \Edge[](n_1)(n_2)
      \Edge[](n_3)(n_4)
    \end{tikzpicture}
\end{center}
  What is the complement of $G$?
\end{problem}
